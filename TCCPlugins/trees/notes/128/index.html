:: TestDBF.btm

:: This tests the functions from the dbf64.dll plugin

::

@setlocal

@echo off

if plugin dbf plugin /u dbf


plugin /l dbf64.dll

plugin /i dbf


iff plugin dbf then

  :: This is just to remove from memory any same-named functions

  if isfunction dbfversion     unfunction dbfversion

  if isfunction RecCount       unfunction RecCount

  if isfunction BytesInHeader  unfunction BytesInHeader

  if isfunction FieldCount     unfunction FieldCount

  if isfunction FieldDecimal   unfunction FieldDecimal

  if isfunction FieldName      unfunction FieldName

  if isfunction FieldWidth     unfunction FieldWidth

  if isfunction FirstRecOffset unfunction FirstRecOffset

  if isfunction LUpdate        unfunction LUpdate

  if isfunction Recnum         unfunction Recnum

  if isfunction Record         unfunction Record

  if isfunction Recsize        unfunction Recsize

  echo.

  if %# gt 0 (set thedbf=%1) else (set thedbf=e:\utils\trans.dbf)

  iff not exist %thedbf then

    echo USAGE: %_batchname YourFile.dbf

       quit

  endiff

  echo DBFVersion    : %@dbfversion[%thedbf]

  echo RecCount      : %@RecCount[%thedbf]

  echo BytesInHeader : %@BytesInHeader[%thedbf]

  echo FieldCount    : %@FieldCount[%thedbf]

  echo FirstRecOffset: %@FirstRecOffset[%thedbf]

  echo Last Update   : %@LUpdate[%thedbf]


  if exist %temp\results.txt del /q %temp\results.txt

  iff exist %thedbf then

    echo Searching for "Harvey's" in %thedbf

    ffind /t"Harvey's" /x %thedbf > %temp\results.txt

    echo _ffind_matches: %_ffind_matches

    iff %_ffind_matches gt 0 then

      :: Get the offset in the file where the search word was found

         set Offset=%@word[1,%@line[%temp\results.txt 2]]

      echo Offset %Offset    : Record # %@Recnum[%thedbf,%Offset]

         echo Record %@Recnum[%thedbf,%Offset]:  %@record[%thedbf,%@Recnum[%thedbf,%Offset]]

    else

      echo Match not found

    endiff

  endiff


  echo Recsize       : %@Recsize[%thedbf]

  echo.

::setdos /x-5

::do kount=1 to %@RecCount[%thedbf]

  set kount=%@RecCount[%thedbf]

  set kount=%@int[%kount/2]

  echo Record %kount : %@Record[%thedbf,%kount]

  echo.

::enddo

::setdos /x+5

  echo Field  Field Name  Type        Width   Dec

  do FieldNumber=1 to %@FieldCount[%thedbf]

    screen %_row 0 %@format[5,%FieldNumber]

    screen %_row 7 %@FieldName[%thedbf,%FieldNumber]

       

       set FieldType=%@FieldType[%thedbf,%FieldNumber]

    switch %FieldType

      case C

        set TheType=Character

         case D

           set TheType=Date

         case L

           set TheType=Logical

         case M

           set TheType=Memo

         case N

           set TheType=Numeric

    endswitch

    screen %_row 19 %TheType

       set FieldWidth=%@FieldWidth[%thedbf,%FieldNumber]

       screen %_row 33 %@format[3,%FieldWidth]

       iff %@FieldType[%dbfName %FieldNumber] eq N then

      screen %_row 40 %@format[2,%@FieldDecimal[%thedbf,%FieldNumber]]

    endiff

       echo.

  enddo

  echo.

  file %thedbf

else

  echo dbf plugin not loaded into memory.

endiff

endlocal
