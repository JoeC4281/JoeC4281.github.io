    Once loaded into memory, any image (whatever the disk file format), could be turned into a Bitmap, to manipulate/examine the DIBBITS.
    In GDImage all images are converted into DIB with the correct x, y origine at 0,0 ready to fullfill an array.
    The GDImage API to do this is
    zGetDIBits(IN HBITMAP hBitmap, LPVOID lpArray);

    or in PB
    zGetDIBits Previous Top Next
    Low level procedure to retrieve a device-independent bitmap.

    FUNCTION zGetDIBits ( _
    BYVAL hBitmap AS LONG, _ ' A valid bitmap handle.
    PixelArray() AS LONG _ ' Array using 2 dimensions (X,Y) matching the pixel coordinates.
    ) AS LONG

    Return:
    If the function succeeds, the return value is the number of scan lines copied.
    If the function fails, the return value is zero.


    and to create the bitmap from any image file format, use this API
    ZI_CreateBitmapFromFile Previous Top Next
    Create a valid GDI BITMAP handle from FILE using any of the supported GDImage format.

    FUNCTION ZI_CreateBitmapFromFile ( _
    zFullImagePath AS ASCIIZ, _ ' The name of an image file.
    BitmapWidth AS LONG, _ ' The width of the bitmap.
    BitmapHeight AS LONG _ ' The Height of the bitmap.
    ) AS LONG

    Return:
    A valid GDI bitmap handle (do not confuse GDI bitmap handle and GDI+ Image handle).

    Important:
    You must use DeleteObject(GDIBitmapHandle) when you do not need it anymore.
    Patrice Terrier
    www.zapsolution.com

                          To answer Marks' query about displaying a PNG File stored as a resource (also works for JPG etc)
                          Uses Jose Roca's or standard PB "win32api.inc". ( Set value of %RocaIncludes)

                          '
                          Code:
                          #COMPILE EXE
#DIM ALL
#DEBUG ERROR ON
#DEBUG DISPLAY ON
#INCLUDE "win32api.inc"

#RESOURCE RCDATA, #1, "MY_IMAGE_A.PNG"

%RT_RCDATA = 10 'Resource Type
%RocaIncludes = 0 ' 0 for PB includes, -1 for Jose's includes

'================ GDI requirements ======================
TYPE GdiplusStartupInput
    GdiplusVersion AS DWORD
    DebugEventCallback AS DWORD
    SuppressBackgroundThread AS LONG
    SuppressExternalCodecs AS LONG
END TYPE

TYPE GdiplusStartupOutput
    NotificationHook AS DWORD
    NotificationUnhook AS DWORD
END TYPE

'GDI+ Declares
DECLARE FUNCTION GdiplusStartup LIB "GDIPLUS.DLL" ALIAS "GdiplusStartup" (token AS DWORD, inputbuf AS GdiplusStartupInput, outputbuf AS GdiplusStartupOutput) AS LONG
DECLARE SUB GdiplusShutdown LIB "GDIPLUS.DLL" ALIAS "GdiplusShutdown" (BYVAL token AS DWORD)
DECLARE FUNCTION GdipGetImageDimension LIB "GDIPLUS.DLL" ALIAS "GdipGetImageDimension" (BYVAL nImage AS LONG, nWidth AS SINGLE, Height AS SINGLE) AS LONG
DECLARE FUNCTION GdipCreateFromHDC LIB "GDIPLUS.DLL" ALIAS "GdipCreateFromHDC" (BYVAL hdc AS LONG, graphics AS LONG) AS LONG
DECLARE FUNCTION GdipDisposeImage LIB "GDIPLUS.DLL" ALIAS "GdipDisposeImage" (BYVAL lpImage AS DWORD) AS LONG
DECLARE FUNCTION GdipDrawImageRectI LIB "GDIPLUS.DLL" ALIAS "GdipDrawImageRectI" (BYVAL hgraphics AS DWORD, BYVAL nimage AS DWORD, BYVAL x AS LONG, BYVAL y AS LONG, _
BYVAL nWidth AS LONG, BYVAL nWeight AS LONG) AS LONG
DECLARE FUNCTION GdipDeleteGraphics LIB "GDIPLUS.DLL" ALIAS "GdipDeleteGraphics" (BYVAL Graphics AS DWORD) AS LONG

#IF %RocaIncludes
    DECLARE FUNCTION GdipCreateBitmapFromStream LIB "GDIPLUS.DLL" ALIAS "GdipCreateBitmapFromStream" (BYVAL pstream AS iStream, m_pbitmap AS LONG) AS LONG
#ELSE
    DECLARE FUNCTION GdipCreateBitmapFromStream LIB "GDIPLUS.DLL" ALIAS "GdipCreateBitmapFromStream" (BYVAL pstream AS DWORD, m_pbitmap AS LONG) AS LONG
    DECLARE FUNCTION CreateStreamOnHGlobal LIB "OLE32.DLL" ALIAS "CreateStreamOnHGlobal" (BYVAL hglobal AS DWORD, BYVAL fdeleteonrelease AS DWORD, pstm AS DWORD) AS DWORD
#ENDIF


FUNCTION PBMAIN
LOCAL hWin, gdiToken AS DWORD
LOCAL Inputbuf AS GdiplusStartupInput

GRAPHIC WINDOW NEW " ",150,100,600,600 TO hWin
GRAPHIC ATTACH hWin, 0, REDRAW

Inputbuf.GdiplusVersion = 1
IF GdiplusStartup(gdiToken, Inputbuf, $NUL) = 0 THEN
    IF GRAPHIC(DC)<> 0 THEN
        DispResource("#1",0,0,1)
    END IF
    GdiplusShutdown gdiToken
END IF

GRAPHIC WAITKEY$
GRAPHIC WINDOW END
END FUNCTION

SUB DispResource(ResID AS ASCIIZ, xpos AS LONG,ypos AS LONG,ScaleFact AS DOUBLE)
'load image stored as RCData into currently attached PB Graphics target
LOCAL hResource, ImageSize, pResourceData, m_hBuffer, pBuffer, pImage,pGraphicObj,hDC AS DWORD
LOCAL W, H AS SINGLE
#IF %RocaIncludes
    LOCAL pStream AS iStream
#ELSE
    LOCAL pStream AS DWORD
#ENDIF
'Get resource
hResource = FindResourceA(0,ResID, BYVAL %rt_rcdata)
imageSize = SizeOfResource(0, hResource)
pResourceData = LockResource(LoadResource(0, hResource))

'Get some memory to use
GLOBALMEM ALLOC imageSize TO m_hBuffer
GLOBALMEM LOCK m_hBuffer TO pBuffer

'Convert resource data to BMP
MEMORY COPY pResourceData, pBuffer, imageSize
CreateStreamOnHGlobal(m_hBuffer, 0, pStream)
GdipCreateBitmapFromStream(pStream, pImage)
GdipGetImageDimension(pImage,W,H)

'Put it in Graphic target
GRAPHIC GET DC TO hDC
GdipCreateFromHDC(hDC, pGraphicObj)
GdipDrawImageRectI(pGraphicObj, pImage, xpos, ypos, ScaleFact*W, ScaleFact*H)

'cleanup memory
GLOBALMEM FREE m_hBuffer TO pBuffer
GdipDeleteGraphics(pGraphicObj)
GdipDisposeImage(pImage)

' refresh image
GRAPHIC REDRAW

END SUB
'
