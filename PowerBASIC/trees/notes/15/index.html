    Input in Graphics Screen - PBCC 6.0

    15 Apr 2011, 05:34 PM
    [Warning:Naive programming question]

    I am in a GRAPHICS screen and I want to check for a key input (normal and function key) or a mouse click. So I go:

    Code:
    DO UNTIL z&<> 0  OR LEN(a$)<>0
    GRAPHIC WINDOW CLICK TO z&,x&,y&
    a$ = GRAPHIC$(INKEY$)
    LOOP
GRAPHIC PRINT z&
GRAPHIC PRINT a$
GRAPHIC PRINT "Key or mouse click"
    Then I analyse z& and a$ to work out what actually happened.

    Is this the obvious way of doing it? Is there any other easy way?

    I am probably only interested in answers at the GOTO level of programming!!!

#COMPILE EXE
#CONSOLE OFF
#DIM ALL
 
FUNCTION PBMAIN () AS LONG
 LOCAL CarH,CarW AS DOUBLE
 LOCAL x1,y1,x2,y2,hWin,fnt1,J,I,PixW,PixH,WinX,WinY AS LONG
 LOCAL CH$,clk&,mx!,my!
 DESKTOP GET CLIENT TO PixW,PixH
 x1=0
 y1=0
 x2=PixW
 y2=PixH
 DESKTOP GET SIZE TO WinX,WinY
 GRAPHIC WINDOW "Test",x1,y1,x2,y2 TO hwin
 GRAPHIC ATTACH hwin,0,REDRAW
 GRAPHIC CLEAR &HDCF5F5
 GRAPHIC SET LOC -1,0
 GRAPHIC WIDTH 1
 FONT NEW "Courier New",20,1,0,0,0 TO fnt1
 GRAPHIC SET FONT fnt1
 GRAPHIC CHR SIZE TO CarW,CarH
 GRAPHIC COLOR %GRAY,&HDCF5F5
 FOR J=1 TO 20
  GRAPHIC SET POS(100,100+J*CarH)
  GRAPHIC PRINT "BACKGROUND BACKGROUND BACKGROUND BACKGROUND BACKGROUND BACKGROUND"
 NEXT
 GRAPHIC SET POS(100,100+CarH*22)
 GRAPHIC PRINT "Use cursor keys to move around or mouse clicks"
 GRAPHIC SET POS(100,100+CarH*23)
 GRAPHIC PRINT "to set the 'cursor' position.....<Esc> to exit"
 GRAPHIC REDRAW
 J=1
 I=0
 DO
  GRAPHIC SET POS(100+11*CarW*I,100+CarH*J)
  GRAPHIC COLOR %BLACK,&H87B8DE
  GRAPHIC PRINT "BACKGROUND"
  GRAPHIC BOX(100+11*CarW*I,100+CarH*J)-(100-CarW+11*CarW*(I+1),100+CarH*(J+1)),,%RED
  GRAPHIC REDRAW
  DO
   GRAPHIC INKEY$ TO CH$
   GRAPHIC WINDOW CLICK TO clk&,mx!,my!
   IF CH$<>"" OR clk&>0 THEN EXIT LOOP
  LOOP
  GRAPHIC SET POS(100+11*CarW*I,100+CarH*J)
  GRAPHIC COLOR %GRAY,&HDCF5F5
  GRAPHIC PRINT "BACKGROUND"
  GRAPHIC REDRAW
  SELECT CASE CH$
   CASE CHR$(0,80)
    INCR J
    IF J>20 THEN J=1
   CASE CHR$(0,72)
    DECR J
    IF J<1 THEN J=20
   CASE CHR$(0,77)
    INCR I
    IF I>5 THEN
     I=0
     INCR J
     IF J>20 THEN J=1
    END IF
   CASE CHR$(0,75)
    DECR I
    IF I<0 THEN
     I=5
     DECR J
     IF J<1 THEN J=20
    END IF
   CASE CHR$(27)
    EXIT LOOP
  END SELECT
  IF (mx!>100 AND mx!<(100+66*CarW)) AND (my!>(100+CarH) AND my!<(100+21*CarH)) THEN
   I=(mx!-100)\11*CarW
   J=(my!-100)\CarH
  END IF
 LOOP
 GRAPHIC WINDOW END
END FUNCTION

                  Here's some test code for exploring key presses and mouse clicks in a Graphic Window that you might like to try.
                  Code:
                  'PBCC60
#CONSOLE OFF
#DIM ALL
 
FUNCTION ControlCode(sKey AS STRING) AS STRING
 FUNCTION = READ$(ASC(sKey))
 
 DATA 01 SOH ^A, 02 STX ^B, 03 ETX ^C, 04 EOT ^D, 05 ENQ ^E, 06 ACK ^F, 07 BEL ^G, 08 BS ^H, 09 HT ^I
 DATA 10 LF ^J, 11 VT ^K, 12 FF ^L, 13 CR ^M, 14 SO ^N, 15 SI ^O, 16 DLE ^P, 17 DC1 ^Q, 18 DC2 ^R
 DATA 19 DC3 ^S, 20 DC4 ^T, 21 NAK ^U, 22 SYN ^V, 23 ETB ^W, 24 CAN ^X, 25 EM ^Y, 26 SUB ^Z
 DATA 27 ESC ^[, 28 FS ^\, 29 GS ^], 30 RS ^^, 31 US ^_
END FUNCTION
'------------------/ControlCode
 
FUNCTION PBMAIN()
 LOCAL hGWnd, hFont AS DWORD
 LOCAL sKey, sTemp AS STRING
 LOCAL Clk AS LONG
 LOCAL x, y AS SINGLE
 
  FONT NEW "Courier New", 12 TO hFont
  GRAPHIC WINDOW TEXT "KeyMon", 200, 300, 20, 50, hFont TO hGWnd
  GRAPHIC ATTACH hGWnd, 0
  GRAPHIC CELL = 10,0 : GRAPHIC PRINT "  Press any key - result displayed in caption"
 
  DO WHILE GRAPHIC(DC) <> 0                         ' Exit loop when hGWnd closes
    GRAPHIC INKEY$ TO sKey
    SELECT CASE LEN(sKey)
      CASE 0                                        ' No keys pressed
 
      CASE 1                                        ' ASCII Key pressed
        GRAPHIC CELL = 5,1 : GRAPHIC PRINT POS(10) sKey
        GRAPHIC SET CAPTION " " + sKey + " =" + STR$(ASC(sKey)) + ", &H" + HEX$(ASC(RIGHT$(sKey,1)))
        IF ASC(sKey) => 1 AND ASC(sKey) =< 31 THEN  ' Control Code
          sTemp = ControlCode(sKey)
          GRAPHIC SET CAPTION "Control Code: " + sTemp
        END IF
        IF sKey = $SPC THEN
          GRAPHIC SET CAPTION " <space> =" + STR$(ASC(sKey)) + ", &H" + HEX$(ASC(RIGHT$(sKey,1)))
        END IF
        sKey = ""
 
      CASE 2                                        ' Extended key pressed.
        SELECT CASE ASC(RIGHT$(sKey,1))
          CASE 59 : sTemp = "F1"
          CASE 60 : sTemp = "F2"
          CASE 61 : sTemp = "F3"
          CASE 62 : sTemp = "F4"
          CASE 63 : sTemp = "F5"
          CASE 64 : sTemp = "F6"
          CASE 65 : sTemp = "F7"
          CASE 66 : sTemp = "F8"
          CASE 67 : sTemp = "F9"
          CASE 68 : sTemp = "F10"
          CASE 87 : sTemp = "F11"                   ' Since Ver. 903/503
          CASE 88 : sTemp = "F12"                   ' Since Ver. 903/503
          CASE 69 : sTemp = "NumLock"
          CASE 71 : sTemp = "Home"                  ' * NB  These keys not detected on number pad
          CASE 72 : sTemp = "Up Arrow"              '       prior to Ver. 904/504*
          CASE 73 : sTemp = "Page up"               '
          CASE 75 : sTemp = "Left arrow"            '
          CASE 77 : sTemp = "Right arrow"           '
          CASE 79 : sTemp = "End"                   '
          CASE 80 : sTemp = "Down arrow"            '
          CASE 81 : sTemp = "Page down"             '
          CASE 82 : sTemp = "Insert"                '
          CASE 83 : sTemp = "Delete"                '
          CASE 91, 92 : sTemp = "WindowsKey"
          CASE ELSE : sTemp = "Other:" + STR$(ASC(RIGHT$(sKey,1)))
        END SELECT
        GRAPHIC SET CAPTION " " + sTemp + " =" + STR$(ASC(RIGHT$(sKey,1))) + ", &H" + HEX$(ASC(RIGHT$(sKey,1)))
 
        sTemp = ""
 
    END SELECT
 
    ' Test for mouse clicks..
    Clk = 0
    GRAPHIC WINDOW CLICK hGWnd TO Clk, x, y
    SLEEP 10
    GRAPHIC WINDOW CLICK hGWnd TO Clk, x, y
    IF Clk = 2 THEN
      GRAPHIC SET CAPTION "Double-Click"
    ELSEIF Clk = 1 THEN
      GRAPHIC SET CAPTION "Click"
      GRAPHIC SET POS (0, 0) : GRAPHIC PRINT "                    " ' clear old pos info
      GRAPHIC SET POS (0, 0)
      GRAPHIC PRINT " x:"+STR$(x)+", y:"+STR$(y)    ' click position in GfxWindow
    END IF
 
  LOOP
 
END FUNCTION
'------------------/PBMain
                  Rgds, Dave

                    GRAPHIC WINDOW TEXT make it easy to resize window

                    I just rewrote Manuel's program to show how easy the new TEXT mode make a changing window size option a snap.

                    Just hit the + and - key to check the effect.

                    My Graphic Console code was much more complicated than that.

                    Many thanks to Bob for that nice gift.
                    Code:
                    ' GraphicInput By Manuel Valdez and modified by Guy Dombrowski
#COMPILE EXE
#CONSOLE OFF
#BREAK ON
#DIM ALL

FUNCTION PBMAIN () AS LONG
 LOCAL x0, y0, x1, y1, CarH, carW, hWin, fnt1, PixW, PixH, nCol, nRow AS LONG
 LOCAL CH$, clk&, mx!, my!, pt!
 DESKTOP GET CLIENT TO PixW, PixH
 pt!=10 ' Initial font size

 GOSUB MakeWindow ' Build initial window

 DO UNTIL GRAPHIC(DC)=0 ' Window closed with [x]

   DO UNTIL GRAPHIC(DC)=0
     GRAPHIC INKEY$ TO CH$: SLEEP 1: GRAPHIC WINDOW CLICK TO clk&, mx!, my!
   LOOP UNTIL CH$<>"" OR clk&<>0

   SELECT CASE CH$
     CASE CHR$(0,80) ' Down arrow
       IF nRow<20 THEN INCR nRow ELSE BEEP
     CASE CHR$(0,72) ' Up arrow
       IF nRow>1 THEN DECR nRow ELSE BEEP
     CASE CHR$(0,77) ' Right arrow
       IF nCol<6 THEN INCR nCol ELSE BEEP
     CASE CHR$(0,75) ' Left arrow
       IF nCol>0 THEN DECR nCol ELSE BEEP
     CASE CHR$(43)   ' + key
       INCR pt!: GOSUB MakeWindow
     CASE CHR$(45)   ' - key
       DECR pt!: GOSUB MakeWindow
     CASE CHR$(27)   ' Esc key
       EXIT LOOP
   END SELECT

   IF (mx!>CarW*2 AND mx!<(CarW*78)) AND (my!>(CarH) AND my!<(CarH*21)) THEN
     nCol=((mx!-carW)\(11*CarW)): nRow=(my!\CarH)
   END IF

   GRAPHIC SET POS((CarW*2)+(CarW*nCol*11),(CarH*nRow)): GRAPHIC COLOR %BLACK, %YELLOW
   GRAPHIC PRINT "BACKGROUND": GRAPHIC REDRAW: SLEEP 200
   GRAPHIC SET POS((CarW*2)+(CarW*nCol*11),(CarH*nRow)): GRAPHIC COLOR %GRAY, &HDCF5F5
   GRAPHIC PRINT "BACKGROUND": GRAPHIC REDRAW
 LOOP

 GRAPHIC WINDOW END: END

MakeWindow: ' Make text window with 25 lines of 80 characters based upon font size
 GRAPHIC WINDOW END
 FONT NEW "Courier New",pt!,1,0,0,0 TO fnt1: GRAPHIC SET FONT fnt1
 GRAPHIC WINDOW TEXT "Testing the new GRAPHIC WINDOW TEXT Command",x1, y1, 25, 80, fnt1 TO hwin
 GRAPHIC ATTACH hwin,0,REDRAW: GRAPHIC CLEAR &HDCF5F5
 GRAPHIC CELL SIZE TO CarW, CarH: GRAPHIC COLOR %GRAY, &HDCF5F5
 GRAPHIC SET LOC 0, 0: GRAPHIC WIDTH 1
   FOR nRow=1 TO 20
     GRAPHIC SET POS(CarW*2,nRow*CarH):  GRAPHIC PRINT "BACKGROUND BACKGROUND BACKGROUND BACKGROUND BACKGROUND BACKGROUND BACKGROUND"
   NEXT
 GRAPHIC SET POS(CarW*2,CarH*22): GRAPHIC PRINT "Use cursor keys to move around or mouse clicks to set the Cursor position"
 GRAPHIC SET POS(CarW*2,CarH*23): GRAPHIC PRINT "Use + and - keys to increase or decrease window size.  <esc> or [x] to end"
 GRAPHIC REDRAW
 nCol=0: nRow=1
 RETURN

END FUNCTION
                    Old QB45 Programmer

 GLOBAL fnt1, PixW, PixH, nCol, nRow, hGC AS LONG

FUNCTION StaticProc(BYVAL hWin AS DWORD, BYVAL Msg AS DWORD, BYVAL wParam AS DWORD, BYVAL lParam AS LONG) AS LONG
  SELECT CASE AS LONG Msg
    CASE %WM_USER + 1000: CreateCaret hWin, 0, CarW, CarH  ' Create solid caret same size as font character grid
    CASE %WM_USER + 1001                                   ' User message to update caret pos
      cx = LO(WORD, lParam) : cy = HI(WORD, lParam): SetCaretPos cx, cy: ShowCaret hWin
    CASE %WM_KILLFOCUS
      DestroyCaret ' Don't show caret if focus elsewhere
    CASE %WM_SETFOCUS
      CreateCaret hWin, 0, CarW, CarH:  SetCaretPos cx, cy: ShowCaret hWin  ' Focus is back! - show caret
    CASE %WM_CHAR
      IF LO(WORD,wParam)=9 THEN tabkey%=9 ' Catch the Tab key
  END SELECT
 FUNCTION = CallWindowProc(GetProp (GetParent(hWin), "OldStaticProc"), hWin, Msg, wParam, lParam)
END FUNCTION

FUNCTION GraphicProc(BYVAL hWnd AS DWORD, BYVAL wMsg AS DWORD, BYVAL wParam AS DWORD, BYVAL lParam AS LONG) AS LONG
  SELECT CASE AS LONG wMsg
  CASE %WM_SETCURSOR  ' If ghCur = 0 mouse pointer will not show at all
    IF GetCursor <> ghCur THEN SetCursor ghCur
    FUNCTION = 1 : EXIT FUNCTION
  END SELECT
  FUNCTION = CallWindowProc (gOldGraphicProc, hWin, wMsg, wParam, lParam)
END FUNCTION

FUNCTION CallBackProc( BYVAL hWin AS DWORD, BYVAL wMsg AS DWORD,BYVAL wParam AS DWORD, BYVAL lParam AS LONG) AS LONG
   SELECT CASE wMsg
     CASE %WM_COMMAND
     CASE %WM_DESTROY: exitprocess 0
     CASE ELSE : FUNCTION = CallWindowProc(OldProc, hWin, wMsg, wParam, lParam)
   END SELECT
END FUNCTION

#COMPILE EXE

DECLARE FUNCTION GetAsyncKeyState LIB "User32.dll" ALIAS 
"GetAsyncKeyState"(BYVAL vKey AS LONG) AS INTEGER  'Dave Biggs hint
 
%VK_SHIFT = &H10
 
GLOBAL ASTR$(),INPT$()
GLOBAL CarH,CarW AS DOUBLE
GLOBAL clk&,mx!,my!
 
FUNCTION PBMAIN () AS LONG
 LOCAL x1,y1,x2,y2,hWin,fnt1,J,I,E,F,PixW,PixH,WinX,WinY AS LONG
 LOCAL CH$,CAD$,CRS%,Vshift%

 x1=0

 x2=PixW
 y2=PixH
 GRAPHIC WINDOW "Test",x1,y1,x2,y2 TO hwin

 GRAPHIC SET POS(100,100+CarH*23)
 GRAPHIC PRINT "Use Cursor or Shift Cursor keys to move around, or mouse clicks"
 GRAPHIC SET POS(100,100+CarH*24)

 GRAPHIC REDRAW
 E=0
 F=1
 DO
  IF GRAPHIC(DC)=0 THEN EXIT LOOP
  SLEEP 1

   FOR I=0 TO 5

    GRAPHIC PRINT LEFT$(INPT$(I,J)+STRING$(10,32),10)
   NEXT
  NEXT

    E=(mx!-100)\(10*CarW+1)

   CAD$=INPT$(E,F)
   CALL FORMATTED_INPUT_X(101+10*CarW*E+E,101+CarH*F+F,10,9,CAD$,1,1,0,CRS%)
   SELECT CASE CRS%

     IF E<5 THEN
      INCR E
     ELSE

       E=0
       INCR F
      END IF
     END IF
    CASE 2

    CASE 5

SUB FORMATTED_INPUT_X(XF AS DOUBLE,YF AS DOUBLE ,LF AS LONG,LC AS LONG,CAD$,PF AS LONG,FT AS LONG,DEC AS LONG,CRS%)
 REDIM ASTR$(2*LF)
 LOCAL CD$,CH1 AS STRING
 LOCAL L,PI,I AS LONG
 LOCAL CH$,Vshift%
 ASTR$(0)=CHR$(0)
 CD$=CAD$+STRING$(LF,32)
 FOR I = 1 TO LF
  ASTR$(I) = MID$(CD$,I,1)
 NEXT
 PI=1
 L = LF
 WHILE ASTR$(L) = CHR$(32)
  DECR L
 WEND
 IF L < 0 THEN L = 0
 IF L=0 THEN
  '
 ELSEIF L>LC THEN
  PI=L-LC
 ELSE
  '
 END IF
 IF PF=0 THEN PF=1
 DO
  IF GRAPHIC(DC)=0 THEN EXIT LOOP
  IF PF < PI THEN DECR PI
  IF PF > PI + LC THEN INCR PI
  GRAPHIC SET POS(XF,YF)
  CH1$ = CHR$(0)
  FOR I = PI TO PI + LC
   IF I = PF THEN
    GRAPHIC COLOR %WHITE,RGB(87,118,173)
   ELSE
    GRAPHIC COLOR %WHITE,&H20A5DA  '&H5C5CCD
   END IF
   IF FT = 5 THEN
    GRAPHIC PRINT CHR$(176);
   ELSE
    GRAPHIC PRINT ASTR$(I);
   END IF
   IF FT=2 THEN
    IF I=2 OR I=4 THEN
     GRAPHIC PRINT CHR$(47);
    END IF
   END IF
  NEXT
  GRAPHIC REDRAW
  CH$=""
  DO
   SLEEP 0
   IF GRAPHIC(DC)=0 THEN EXIT LOOP
   GRAPHIC WINDOW CLICK TO clk&, mx!, my!
   GRAPHIC INKEY$ TO CH$
   Vshift% = GetAsyncKeyState(%VK_SHIFT)     'Dave Biggs hint
   IF CH$<>"" OR clk& THEN
    IF clk& THEN CH$=CHR$(0)
    EXIT LOOP
   END IF
  LOOP
  IF clk& THEN EXIT LOOP
  IF LEN(CH$) = 1 THEN
   SELECT CASE CH$
    CASE CHR$(8)
     FOR I = PF TO L
      ASTR$(I-1) = ASTR$(I)
     NEXT
     IF PF > 1 THEN DECR PF
     IF L > 1 THEN
      ASTR$(L) = CHR$(32)
      DECR L
     END IF
    CASE CHR$(9)
     EXIT SELECT
    CASE CHR$(27)
     EXIT SELECT
    CASE  > CHR$(30)
     IF FT=3 AND CH$=CHR$(46) THEN
      IF PF<LF-DEC AND PF>1 THEN
       FOR I=(LF-DEC-1) TO 1 STEP -1
        ASTR$(I)=ASTR$(PF-1)
        ASTR$(PF-1)=CHR$(32)
        DECR PF
        IF PF=1 THEN EXIT FOR
       NEXT
      ELSE
      END IF
      IF DEC>0 THEN ASTR$(LF-DEC)=CHR$(46)
      FOR I=1+LF-DEC TO LF
       ASTR$(I)=CHR$(48)
      NEXT
      L=L+1+DEC
      PF=LF-DEC+1
     ELSE
      IF L < LF THEN INCR L
      FOR I = L TO PF STEP - 1
       ASTR$(I) = ASTR$(I-1)
      NEXT
      ASTR$(PF) = CH$
      GRAPHIC SET POS(8+CarW*(XF+PF-PI&),9+CarH*YF)
      IF PF < LF THEN
       GRAPHIC PRINT CH$;
       INCR PF
       CH$=CHR$(0)
      END IF
     END IF
     IF FT=5 THEN CH$=CHR$(13)
     CH$=CHR$(0,0)
   END SELECT
  ELSE
   CH$=RIGHT$(CH$,1)
   SELECT CASE CH$
    CASE CHR$(75)
     IF Vshift%=0 THEN        '(Shift Key Pressed) Dave Biggs hint
      IF PF > 1 THEN
       DECR PF
      END IF
     ELSE
      CH$=CHR$(117)
     END IF
    CASE CHR$(77)
     IF Vshift%=0 THEN       ' (Shift Key Pressed) Dave Biggs hint
      IF PF < LF THEN
       INCR PF
       IF PF > L THEN INCR L
      END IF
     ELSE
      CH$=CHR$(116)
     END IF
    CASE CHR$(83)
     FOR I = PF TO L-1
      ASTR$(I) = ASTR$(I+1)
     NEXT
     IF L >= PF THEN
      ASTR$(L) = CHR$(32)
      DECR L
     END IF
    CASE ELSE
     IF L < LF THEN INCR L
     FOR I = L TO PF STEP - 1
      ASTR$(I) = ASTR$(I-1)
     NEXT
     GRAPHIC SET POS(XF+CarW*(PF-PI),YF)
     IF PF < LF THEN
      IF FT=5 THEN
       GRAPHIC PRINT CHR$(176)
      ELSE
       'GRAPHIC PRINT CH$
      END IF
      INCR PF
     END IF
   END SELECT
   GRAPHIC REDRAW
  END IF
  '             1-Cr     2-Esc    3-AUp    4-PgUp   5-ADn    6-PgDn   7-F10   8-S_ALeft 9-S_ARight 10-^PgDn
  CRS% = INSTR(CHR$(13)+CHR$(27)+CHR$(72)+CHR$(73)+CHR$(80)+CHR$(81)+CHR$(66)+CHR$(116)+CHR$(117)+CHR$(118)+_
  CHR$(132)+CHR$(71)+CHR$(79)+CHR$(82)+CHR$(60)+CHR$(63)+CHR$(61)+CHR$(59)+CHR$(67)+CHR$(64)+CHR$(65)+CHR$(62),CH$)
  '11-^ALeft 12-Home   13-END   14-Ins   15-F2    16-F5    17-F3    18-F1    19-F9    20-F6    21-F7    22-F4
  CD$ = ""
  IF CRS% = 1 THEN
   FOR I = 1 TO L
    CD$=CD$+ASTR$(I)
   NEXT
   CAD$=CD$
   EXIT LOOP
  ELSEIF CRS%=2 THEN
   EXIT LOOP
  ELSEIF CRS% > 2 THEN
   IF PF>LF THEN DECR PF
   EXIT LOOP
  END IF
 LOOP
END SUB
