vfp.library for use with Take Command Console
VFP {
  library vfp.*
}

VFP.StartServer {
  @setlocal
  @echo off
  pshell /s "$vfp = CreateObject('VisualFoxPro.Application.9')"
  endlocal
}

VFP.DoCmd {
  @echo off
pshell /s "$VFP.DoCmd('%$')" > nul
}

VFP.Eval {
  @setlocal
  @echo off
  pshell /s "$vfp.eval('%$')"
  endlocal
}

VFP.SetVar {
  @setlocal
  @echo off
  pshell /s "$vfp.SetVar('%1',%2)"
  endlocal
}

VFP.StopServer {
  @setlocal
  @echo off
  pshell /s "$vfp.Quit()"
  endlocal
}
