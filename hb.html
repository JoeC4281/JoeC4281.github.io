hb.library for use with Take Command Console
hb {
@setlocal
@echo off
::
:: hb date()
:: hb 2022-1960
::

iff %1 eq help .or. %# eq 0 then
  echo USAGE: hb date()
echo        hb 2022-1960
quit
endiff

:: Disable the Escape Character
setdos /X-8

if exist r:\ (set dest=r:\hb.prg) else (set dest=%temp\hb.prg)

echo Proc Main() > %dest
echo OutStd(%*) >> %dest
echo Return >> %dest

set _hb=%@execstr[e:\hb34\bin\hbrun.exe %dest]

echo %_hb

setdos /X+8

endlocal _hb
}

hbr {
@setlocal
@echo off
::
:: hbr setmode(25,80);use TRNSCTNS.DBF;browse()
::

iff %1 eq help .or. %# eq 0 then
  echo USAGE: hbr setmode(25,80);use TRNSCTNS.DBF;browse()
  echo        hbr local bday;local days;bday=ctod("1962/11/03"); days = date() - bday; ?? days; ? days/365.25

echo hbr setmode(25,80);use TRNSCTNS.DBF;browse() > clip:
keystack LWin-V
quit
endiff

:: Disable the Escape Character
setdos /X-8

if exist r:\ (set dest=r:\hb.prg) else (set dest=%temp\hb.prg)

echo Proc Main() > %dest
echo %* >> %dest
echo Return >> %dest

e:\hb34\bin\hbrun.exe %dest

setdos /X+8

}
NOTES: Drive R: on my system is a RAMDrive

The hb library command returns the result in the environment variable _hb

Thus, hb 30*12 will store the result in _hb.

You can then echo %_hb to see the result.
